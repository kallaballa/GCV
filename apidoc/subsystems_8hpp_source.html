167 bool capture(std::function<void(cv::UMat&)> fn) {
168 {
169 CLExecScope_t scope(context_);
170 fn(videoFrame_);
171 }
172 {
173 CLExecScope_t scope(fbContext_.getCLExecContext());
174 fbContext_.acquireFromGL(frameBuffer_);
175 if (videoFrame_.empty())
176 return false;
177
178 cv::cvtColor(videoFrame_, frameBuffer_, cv::COLOR_RGB2BGRA);
179 cv::Size fbSize = fbContext_.getSize();
180 cv::resize(frameBuffer_, frameBuffer_, fbSize);
181 fbContext_.releaseToGL(frameBuffer_);

183 }
184 return true;
185 }

190 CLExecScope_t scope(fbContext_.getCLExecContext());

194 fbContext_.releaseToGL(frameBuffer_);
195 }

198 CLExecScope_t scope(context_);
199 fn(videoFrame_);

559static std::string get_cl_info() {
560 std::stringstream ss;
561 std::vector<cv::ocl::PlatformInfo> plt_info;
562 cv::ocl::getPlatfomsInfo(plt_info);
563 const cv::ocl::Device &defaultDevice = cv::ocl::Device::getDefault();
564 cv::ocl::Device current;
565 ss << endl;
566 for (const auto &info : plt_info) {
567 for (int i = 0; i < info.deviceNumber(); ++i) {
568 ss << "\t";
569 info.getDevice(current, i);
570 if (defaultDevice.name() == current.name())
571 ss << "* ";
572 else
573 ss << " ";
574 ss << info.version() << " = " << info.name() << endl;
575 ss << "\t\t GL sharing: " << (current.isExtensionSupported("cl_khr_gl_sharing") ? "true" : "false") << endl;
576 ss << "\t\t VAAPI media sharing: " << (current.isExtensionSupported("cl_intel_va_api_media_sharing") ? "true" : "false") << endl;
577 }
578 }
579
580 return ss.str();
581}

588static void update_fps(cv::Ptr<Window> window, bool graphical = false) {
589 static uint64_t cnt = 0;
590 static double fps = 1;
591 static cv::TickMeter meter;
592
593 if (cnt > 0) {
594 meter.stop();
595
596 if (cnt % uint64(ceil(fps)) == 0) {
597 fps = meter.getFPS();
598 cerr << "FPS : " << fps << '\r';
599 cnt = 0;
600 }
601 }
602
603 if (graphical) {
604 window->renderNVG([&](NVGcontext *vg, const cv::Size &size) {
605 string text = "FPS: " + std::to_string(fps);
606 nvgBeginPath(vg);
607 nvgRoundedRect(vg, 10, 10, 30 * text.size() + 10, 60, 10);
608 nvgFillColor(vg, nvgRGBA(255, 255, 255, 180));
609 nvgFill(vg);
610
611 nvgBeginPath(vg);
612 nvgFontSize(vg, 60.0f);
613 nvgFontFace(vg, "mono");
614 nvgFillColor(vg, nvgRGBA(90, 90, 90, 255));
615 nvgTextAlign(vg, NVG_ALIGN_LEFT | NVG_ALIGN_MIDDLE);
616 nvgText(vg, 22, 37, text.c_str(), nullptr);
617 });
618 }
619
620 meter.start();
621 ++cnt;
622 }
